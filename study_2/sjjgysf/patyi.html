# PAT (Basic Level) Practice （中文）

# 1001 还是人不偿命的（3n+1）猜想

分类：简单模拟

思路

  1. 如果 n 为 1，则退出循环
  2. 如果 n 不为 1，则判断 n 是否为偶数，如果是偶数，则令 n 除以 2；否则令 n 为（3*n+1）/2。之后令计数器 step 加 1。
C(gcc) 
#include<stdio.h>

int main(void){
    int n, step;
    step = 0;
    scanf("%d",&n);
    while(n!=1){
        if(n%2==0){
            n=n/2;
        }else{
            n=(3*n+1)/2;
        }
        step++;
    }
    printf("%d",step);
    return 0;
}

# 1032 挖掘机技术哪家强

分类：简单模拟

疑问：如果成绩相等怎么办？

C(gcc) 
#include <stdio.h>
int schoolzongfen[100010]={0};  //记录每个学校的总分

int main(void){
    int i,n,schoolid,fenshu,zuigaofenid = 1,zuigaofenshu = -1;
    scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d%d",&schoolid,&fenshu);
		schoolzongfen[schoolid] += fenshu;  //录入数据
	}
	for(i=1;i<=n;i++){      //i=1 !!! 数组问题
		if(schoolzongfen[i]>zuigaofenshu){
			zuigaofenshu = schoolzongfen[i];
			zuigaofenid = i;
		}
	}
	printf("%d %d\n",zuigaofenid,zuigaofenshu);
	return 0;
}
